CREATE TABLE `listingslatest` (
  `id` int,
  `symbol` text,
  `name` text,
  `volume_24h` float,
  `percent_change_1h` float,
  `percent_change_24h` float,
  `percent_change_7d` float,
  `market_cap` float,
  `price` double,
  `timestamp` bigint,
  PRIMARY KEY (`id`)
);

CREATE TABLE `searches` (
  `numSearch` bigint,
  `id` int,
  `symbol` text,
  `name` text,
  `timestamp` bigint,
  PRIMARY KEY (`numSearch`)
);
